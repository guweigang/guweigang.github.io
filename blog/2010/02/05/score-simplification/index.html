分数化简算法

// @param denominator 分母
// @param numerator 分子
// 化分母为正数
if (denominator < 0)
{
    numerator = -numerator;
    denominator = - denominator;
}

int a = abs(numerator);
int b = abs(denominator);

while( b>0 )
{
    int t  = a % b;
    a = b;
    b = t;
}
numerator /= a;
denominator /= a;
